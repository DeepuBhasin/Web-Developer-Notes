#install apache2 

    Step 1. Sudo apt-cache search apache2                       // to check the package available and check -> apache2 - Apache HTTP Server 
    Step 2. sudo apt update 
    Step 3. sudo apt upgrade
    Step 4. sudo apt-get install apache2
    Step 5. Type y -> enter
    Step 6. sudo systemctl status apache2
    Step 7. sudo ufw allow 80/tcp                               // allow firewall to run service on port 80 
    Step 8. curl localhost                                      // to service of web content and if it load default page of apache2 it means its working fine (incase curl error -> sudo apt-get install curl)    
                
                or 
    
    Step 8 a. hostname -I                                       // return the ip address at where Apache2 is running 
    Step 8 b. type in web browser : 127.0.0.1 or localhost -> if you get apache2 page it means its working successfully 

#apache2 Directory Structure 

    1. configuration Files      : /etc/apache2/sites-available
    2. html files               : /var/www/html

#set Permissions 

    1. Read and write           : sudo chmod -R 755 /var/www/html
    2. Check all Permissions    : ls -la /var/www/html 
 
#Apache2 Commands 

    1. Status   : sudo systemctl status apache2
    2. Start    : sudo systemctl start apache2
    3. Stop     : sudo systemctl stop apache2   
    4. Restart  : sudo systemctl restart apache2    

#create Virtual Host 

    Step 1 a. create folder in /var/www/html/mypractice/basic
    Step 1 b. create folder in /var/www/html/mypractice/mysqlbasic
    


    Step 2. cd /etc/apache2/sites-available/
    step 3 a. sudo cp 000-default.conf basic.com.conf               // setting domain names in file

            <VirtualHost *:80>
               ServerAdmin webmaster@localhost
                ServerName basic.com
                ServerAlias	www.basic.com
                DocumentRoot /var/www/html/mypractice/basic
                <Directory /var/www/html/mypractice/basic >
                        Options Indexes FollowSymLinks MultiViews
                        AllowOverride all
                        Order allow,deny
                        allow from all
                        Require all granted
                </Directory>
            </VirtualHost>


    Step 3 b. sudo cp basic.com.conf mysqlbasic.com.conf                // setting domain name in file

             <VirtualHost *:80>
               ServerAdmin webmaster@localhost
                ServerName mysqlbasic.com
                ServerAlias	www.mysqlbasic.com
                DocumentRoot /var/www/html/mypractice/mysqlbasic
                <Directory /var/www/html/mypractice/mysqlbasic >
                        Options Indexes FollowSymLinks MultiViews
                        AllowOverride all
                        Order allow,deny
                        allow from all
                        Require all granted
                </Directory>
            </VirtualHost>

    Step 6 a. sudo a2ensite basic.com.conf                                        // enable domain name  
    step 6 b. sudo a2ensite mysqlbasic.com.conf                                   // enable domain name   
    step 6 c. sudo a2dissite 000-default.conf

    Step 7. sudo vi /etc/hosts
    Step 8. Add Below Code 

            127.0.0.1   basic.com
            127.0.0.1   mysqlbasic.com 
    
    Step 9. sudo systemctl reload apache2
    Step 10. sudo systemctl status apache2
